public class TabelPerkalian {
    public static void main(String[] args) {
        for (int i = 1; i <= 10; i++) {
            for (int j = 1; j <= 10; j++) {
                int hasil = i * j;
                System.out.printf("%3d  ", hasil);
            }
            System.out.println();
        }
    }
}
